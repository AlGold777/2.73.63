Что происходит сейчас
Проблема в двух словах: Расширение не знает, когда "забыть" про старые задачи.

Сценарий 1: Опоздавший курьер
Ты запустил запрос → расширение отправило команды во вкладки → ты нажал "Стоп" → запустил новый запрос.
Но старые команды уже "в пути". Через 3 секунды они доставляются — и ломают новый запрос, потому что расширение не проверяет "а это вообще для текущей сессии?"

Сценарий 2: Два блокнота
Расширение записывает "какая вкладка = какая модель" в два места одновременно:
    • TabMapManager (медленно, через storage) 
    • jobState.llms[].tabId (быстро, в памяти) 
Они рассинхронизируются. Одна часть кода смотрит в один блокнот, другая — в другой. Хаос.

Сценарий 3: Призрак в браузере
Ты на  .ai нажал "New Chat". URL изменился, страница перерисовалась — но content script этого не заметил. Он всё ещё слушает старые элементы, которых уже нет в DOM. Промпт уходит в никуда.

Как должно быть
1. Номерок сессии
Каждый запуск получает уникальный номер. Любой отложенный callback первым делом спрашивает: "Мой номер ещё актуален?" Нет — молча уходит.
2. Один блокнот
Только TabMapManager знает какая вкладка какой модели принадлежит. Точка. Никаких дублей.
3. Детектор навигации
Content script следит за URL. Изменился pathname? Очистить всё, переинициализироваться.

Race Conditions в LLM_Codex — Экспертный Анализ
Критическая Оценка Предоставленных Анализов
Что переоценено (отбрасываю):
    1.   CAS/SharedArrayBuffer — Overkill. JavaScript single-threaded, настоящих data races в классическом понимании нет. chrome.storage.sync НЕ имеет {ifVersion} API — это выдуманный код.
    2.   "6-8 недель рефакторинга" — Паникёрство. Проблемы решаются за 1-2 недели фокусированной работы.
    3.  : Network-level scraping — Интересно, но это pivot архитектуры, не quick fix. DOM-scraping работает в 90%+ случаев, проблема не в нём.
    4.   Mutex для scroll-toolkit — Content scripts изолированы per-tab, глобальный mutex между вкладками невозможен в принципе (разные контексты исполнения).

Проблема #4: SPA Navigation Zombies (КРИТИЧЕСКАЯ)
Факт: Content scripts не детектируют SPA navigation. При переходе с  .ai/ на  .ai/chat/new:
    • URL меняется 
    • DOM полностью пересоздаётся 
    • Event listeners висят на старых nodes 
    • MutationObservers наблюдают detached trees 
Моё решение — добавить в каждый content script:
// content-*.js — В начало скрипта
(function initSPANavigationGuard() {
    let lastUrl = location.href;
    let lastPathname = location.pathname;
    
    // Способ 1: MutationObserver на URL change
    const urlObserver = new MutationObserver(() => {
        if (location.href !== lastUrl) {
            const oldUrl = lastUrl;
            lastUrl = location.href;
            
            // Если только pathname изменился — это SPA navigation
            if (location.pathname !== lastPathname) {
                lastPathname = location.pathname;
                console.log(`[SPA-GUARD] Navigation detected: ${oldUrl} -> ${lastUrl}`);
                
                // Cleanup старых ресурсов
                if (window.HumanoidCleanup?.createScope) {
                    const scope = window.__currentCleanupScope;
                    if (scope) scope.cleanup('spa-navigation');
                }
                
                // Signal background
                try {
                    chrome.runtime.sendMessage({ 
                        type: 'SPA_NAVIGATION', 
                        oldUrl, 
                        newUrl: lastUrl 
                    });
                } catch (e) {
                    // Extension context invalidated — см. Проблему #5
                }
                
                // Реинициализация через небольшую задержку (дать React отрендериться)
                setTimeout(() => reinitializeContentScript(), 500);
            }
        }
    });
    
    urlObserver.observe(document.body, { childList: true, subtree: true });
    
    // Способ 2: History API interception (более надёжный)
    const originalPushState = history.pushState;
    const originalReplaceState = history.replaceState;
    
    history.pushState = function(...args) {
        originalPushState.apply(this, args);
        window.dispatchEvent(new Event('spa-navigation'));
    };
    
    history.replaceState = function(...args) {
        originalReplaceState.apply(this, args);
        window.dispatchEvent(new Event('spa-navigation'));
    };
    
    window.addEventListener('popstate', () => {
        window.dispatchEvent(new Event('spa-navigation'));
    });
    
    window.addEventListener('spa-navigation', () => {
        // Тот же cleanup + reinit
    });
})();

Факт: После extension reload/update content script продолжает работать, но все chrome.runtime.* падают.

Моё решение:
// content-utils.js — Добавить обёртку
function safeChromeMessage(message, callback) {
    try {
        if (!chrome.runtime?.id) {
            throw new Error('Extension context invalidated');
        }
        chrome.runtime.sendMessage(message, (response) => {
            if (chrome.runtime.lastError) {
                const err = chrome.runtime.lastError.message || '';
                if (err.includes('Extension context invalidated')) {
                    handleContextInvalidation();
                    return;
                }
            }
            if (callback) callback(response);
        });
    } catch (e) {
        if (e.message.includes('Extension context invalidated')) {
            handleContextInvalidation();
        } else {
            throw e;
        }
    }
}

function handleContextInvalidation() {
    console.warn('[ZOMBIE] Extension context invalidated — cleaning up');
    
    // 1. Очистить все ресурсы
    if (window.__currentCleanupScope) {
        window.__currentCleanupScope.cleanup('extension-invalidated');
    }
    
    // 2. Удалить глобальные маркеры
    delete window.__llmCodexVersion;
    delete window.LLMExtension;
    
    // 3. Показать пользователю (опционально)
    console.info('[LLM_Codex] Extension was reloaded. Please refresh this page.');
}

// Проактивная проверка при старте
function checkExtensionContext() {
    try {
        return !!chrome.runtime?.id;
    } catch {
        return false;
    }
}

Патч #3: SPA Navigation Guard
Создать новый файл: spa-navigation-guard.js
/**
 * SPA Navigation Guard
 * Детектирует client-side navigation и реинициализирует content script
 */
(function initSPANavigationGuard() {
    'use strict';
    
    if (window.__spaGuardInitialized) return;
    window.__spaGuardInitialized = true;
    
    let lastUrl = location.href;
    let lastPathname = location.pathname;
    
    const REINIT_DELAY_MS = 500; // Дать React/Vue отрендериться
    
    function handleNavigation(source) {
        if (location.href === lastUrl) return;
        
        const oldUrl = lastUrl;
        const oldPathname = lastPathname;
        lastUrl = location.href;
        lastPathname = location.pathname;
        
        // Только pathname change = SPA navigation
        if (location.pathname === oldPathname) {
            console.log(`[SPA-GUARD] Hash/query change ignored: ${oldUrl} -> ${lastUrl}`);
            return;
        }
        
        console.log(`[SPA-GUARD] Navigation detected (${source}): ${oldUrl} -> ${lastUrl}`);
        
        // 1. Cleanup старых ресурсов
        cleanupOldResources();
        
        // 2. Notify background
        notifyBackground(oldUrl, lastUrl);
        
        // 3. Reinitialize после задержки
        setTimeout(reinitializeScript, REINIT_DELAY_MS);
    }
    
    function cleanupOldResources() {
        // Использовать существующий HumanoidCleanup если есть
        if (window.__currentCleanupScope?.cleanup) {
            try {
                window.__currentCleanupScope.cleanup('spa-navigation');
            } catch (e) {
                console.warn('[SPA-GUARD] Cleanup error:', e);
            }
        }
        
        // Также проверить humanSessionController
        if (window.humanSessionController?.forceHardStop) {
            try {
                window.humanSessionController.forceHardStop('spa-navigation');
            } catch (e) {}
        }
    }
    
    function notifyBackground(oldUrl, newUrl) {
        try {
            if (!chrome.runtime?.id) return; // Extension invalidated
            
            chrome.runtime.sendMessage({
                type: 'SPA_NAVIGATION',
                oldUrl,
                newUrl,
                timestamp: Date.now()
            }).catch(() => {
                // Ignore - extension might be reloading
            });
        } catch (e) {
            // Extension context invalidated
        }
    }
    
    function reinitializeScript() {
        console.log('[SPA-GUARD] Reinitializing content script...');
        
        // Trigger re-initialization event
        window.dispatchEvent(new CustomEvent('llm-codex-reinit', {
            detail: { url: location.href, reason: 'spa-navigation' }
        }));
    }

                handleNavigation('mutation');
            }

/**
 * Safe Chrome Runtime Messaging
 * Handles extension context invalidation gracefully
 */

const ExtensionContext = {
    _isValid: true,
    _invalidationCallbacks: [],
    
    /**
     * Check if extension context is still valid
     */
    isValid() {
        if (!this._isValid) return false;
        try {
            return !!chrome.runtime?.id;
        } catch {
            this._handleInvalidation();
            return false;
        }
    },
    
    /**
     * Register callback for when context becomes invalid
     */
    onInvalidated(callback) {
        if (typeof callback === 'function') {
            this._invalidationCallbacks.push(callback);
        }
    },
    
    _handleInvalidation() {
        if (!this._isValid) return; // Already handled
        this._isValid = false;
        
        console.warn('[EXTENSION] Context invalidated - cleaning up');
        
        // Notify all registered callbacks
        this._invalidationCallbacks.forEach(cb => {
            try { cb(); } catch (e) { console.error('[EXTENSION] Cleanup error:', e); }
        });
        
        // Global cleanup
        this._globalCleanup();
    },
    
    _globalCleanup() {
        // Stop any active sessions
        if (window.humanSessionController?.forceHardStop) {
            try { window.humanSessionController.forceHardStop('extension-invalidated'); } catch {}
        }
        
        // Cleanup scopes
        if (window.__currentCleanupScope?.cleanup) {
            try { window.__currentCleanupScope.cleanup('extension-invalidated'); } catch {}
        }
        
        // Remove global markers
        delete window.__llmCodexVersion;
        delete window.__spaGuardInitialized;
        
        // User notification (optional)
        console.info('[LLM_Codex] Extension was reloaded. Please refresh this page for full functionality.');
    }
};

/**
 * Safe wrapper for chrome.runtime.sendMessage
 */
function safeSendMessage(message, callback) {
    if (!ExtensionContext.isValid()) {
        console.warn('[EXTENSION] Cannot send message - context invalidated');
        return false;
    }
    
    try {
        chrome.runtime.sendMessage(message, (response) => {
            if (chrome.runtime.lastError) {
                const err = chrome.runtime.lastError.message || '';
                if (err.includes('Extension context invalidated') || 
                    err.includes('Cannot access chrome.runtime')) {
                    ExtensionContext._handleInvalidation();
                    return;
                }
            }
            if (callback) callback(response);
        });
        return true;
    } catch (e) {
        if (e.message?.includes('Extension context invalidated')) {
            ExtensionContext._handleInvalidation();
        }
        return false;
    }
}

/**
 * Safe wrapper for chrome.runtime.connect
 */
function safeConnect(connectInfo) {
    if (!ExtensionContext.isValid()) {
        return null;
    }
    
    try {
        const port = chrome.runtime.connect(connectInfo);
        
        port.onDisconnect.addListener(() => {
            if (chrome.runtime.lastError?.message?.includes('Extension context invalidated')) {
                ExtensionContext._handleInvalidation();
            }
        });
        
        return port;
    } catch (e) {
        if (e.message?.includes('Extension context invalidated')) {
            ExtensionContext._handleInvalidation();
        }
        return null;
    }
}

// Periodic health check (every 30 seconds)
setInterval(() => {
    ExtensionContext.isValid();
}, 30000);

window.safeSendMessage = safeSendMessage;
window.safeConnect = safeConnect;
