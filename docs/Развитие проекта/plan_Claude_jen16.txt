2. TTLMap для глобальных структур
Проблема: Map'ы растут бесконечно, memory leak после 100+ сессий.
Файл: Создать utils/ttl-map.js
Реализация:
class TTLMap {
  constructor(options = {}) {
    this.ttlMs = options.ttlMs || 30 * 60 * 1000; // 30 минут
    this.maxSize = options.maxSize || 500;
    this.cleanupIntervalMs = options.cleanupIntervalMs || 60 * 1000;
    this._map = new Map();
    this._timestamps = new Map();
    this._startCleanupLoop();
  }

  set(key, value) {
    // Evict oldest if at capacity
    if (this._map.size >= this.maxSize && !this._map.has(key)) {
      this._evictOldest();
    }
    this._map.set(key, value);
    this._timestamps.set(key, Date.now());
    return this;
  }

  get(key) {
    if (!this._map.has(key)) return undefined;
    const ts = this._timestamps.get(key);
    if (Date.now() - ts > this.ttlMs) {
      this.delete(key);
      return undefined;
    }
    // Refresh timestamp on access
    this._timestamps.set(key, Date.now());
    return this._map.get(key);
  }

  delete(key) {
    this._map.delete(key);
    this._timestamps.delete(key);
  }

  has(key) {
    return this.get(key) !== undefined;
  }

  clear() {
    this._map.clear();
    this._timestamps.clear();
  }

  get size() {
    return this._map.size;
  }

  _evictOldest() {
    let oldestKey = null;
    let oldestTs = Infinity;
    for (const [key, ts] of this._timestamps) {
      if (ts < oldestTs) {
        oldestTs = ts;
        oldestKey = key;
      }
    }
    if (oldestKey) this.delete(oldestKey);
  }

  _cleanup() {
    const now = Date.now();
    for (const [key, ts] of this._timestamps) {
      if (now - ts > this.ttlMs) {
        this.delete(key);
      }
    }
  }

  _startCleanupLoop() {
    setInterval(() => this._cleanup(), this.cleanupIntervalMs);
  }

  // Совместимость с Map API
  keys() { return this._map.keys(); }
  values() { return this._map.values(); }
  entries() { return this._map.entries(); }
  forEach(cb) { this._map.forEach(cb); }
}

3. SafeMutex с error propagation
Проблема: Ошибки в dispatch глотаются, caller не знает о failure.
Файл: Создать utils/safe-mutex.js
class SafeMutex {
  constructor() {
    this._queue = Promise.resolve();
  }

  async acquire(fn) {
    let release;
    const gate = new Promise(resolve => { release = resolve; });
    
    const prev = this._queue;
    this._queue = gate;
    
    await prev;
    
    try {
      const result = await fn();
      return result;
    } finally {
      release();
    }
  }
}

// Per-model mutex manager
class MutexManager {
  constructor() {
    this._mutexes = new Map();
  }

  async withLock(key, fn) {
    if (!this._mutexes.has(key)) {
      this._mutexes.set(key, new SafeMutex());
    }
    return this._mutexes.get(key).acquire(fn);
  }

  clear(key) {
    this._mutexes.delete(key);
  }
}

const CompressedStorage = {
  async set(key, value) {
    const json = JSON.stringify(value);
    const compressed = LZString.compressToUTF16(json);
    
    // Логировать экономию
    const ratio = ((json.length - compressed.length) / json.length * 100).toFixed(1);
    console.log(`[STORAGE] ${key}: ${json.length} → ${compressed.length} bytes (${ratio}% saved)`);
    
    await chrome.storage.local.set({ [key]: compressed });
  },

  async get(key) {
    const result = await chrome.storage.local.get(key);
    if (!result[key]) return null;
    
    try {
      const decompressed = LZString.decompressFromUTF16(result[key]);
      return JSON.parse(decompressed);
    } catch (e) {
      // Fallback: может быть старые несжатые данные
      console.warn(`[STORAGE] Failed to decompress ${key}, trying raw parse`);
      return typeof result[key] === 'string' ? JSON.parse(result[key]) : result[key];
    }
  },

  async remove(key) {
    await chrome.storage.local.remove(key);
  }
};

const AttachmentHandler = {
  // Определить тип загрузчика на платформе
  detectUploadType(platform) {
    const strategies = {
      GPT: () => document.querySelector('input[type="file"]'),
      Grok: () => document.querySelector('input[type="file"]'),
      Claude: () => document.querySelector('[data-dropzone]') || document.querySelector('.upload-area'),
      Gemini: () => document.querySelector('[role="button"][aria-label*="upload"]'),
      Perplexity: () => document.querySelector('input[accept*="image"]'),
    };
    return strategies[platform]?.();
  },

  // Конвертировать File в DataTransfer
  createDataTransfer(files) {
    const dt = new DataTransfer();
    files.forEach(file => dt.items.add(file));
    return dt;
  },

  // Стратегия для input[type="file"]
  async attachViaInput(input, files) {
    const dt = this.createDataTransfer(files);
    input.files = dt.files;
    input.dispatchEvent(new Event('change', { bubbles: true }));
    await sleep(100);
  },

  // Стратегия для Dropzone (Claude, Gemini)
  async attachViaDropzone(dropzone, files) {
    const dt = this.createDataTransfer(files);
    
    // Симулировать drag-drop последовательность
    const dragEnter = new DragEvent('dragenter', { bubbles: true, dataTransfer: dt });
    const dragOver = new DragEvent('dragover', { bubbles: true, dataTransfer: dt });
    const drop = new DragEvent('drop', { bubbles: true, dataTransfer: dt });
    
    dropzone.dispatchEvent(dragEnter);
    await sleep(50);
    dropzone.dispatchEvent(dragOver);
    await sleep(50);
    dropzone.dispatchEvent(drop);
    await sleep(200);
  },

  // Главный метод
  async attach(platform, files) {
    const element = this.detectUploadType(platform);
    if (!element) {
      console.error(`[ATTACH] No upload element found for ${platform}`);
      return false;
    }

    const isDropzone = element.hasAttribute('data-dropzone') || 
                       element.classList.contains('upload-area') ||
                       element.getAttribute('role') === 'button';

    if (isDropzone) {
      await this.attachViaDropzone(element, files);
    } else {
      await this.attachViaInput(element, files);
    }
    
    return true;
  }
};

8. Retry с exponential backoff + max attempts
Проблема: Supervisor спамит retries бесконечно при degraded модели.
Файл: Создать utils/retry-strategy.js
class RetryStrategy {
  constructor(options = {}) {
    this.maxAttempts = options.maxAttempts || 5;
    this.baseDelayMs = options.baseDelayMs || 1000;
    this.maxDelayMs = options.maxDelayMs || 30000;
    this.jitterFactor = options.jitterFactor || 0.3;
  }

  getDelay(attempt) {
    // Exponential backoff: 1s, 2s, 4s, 8s, 16s...
    const exponentialDelay = this.baseDelayMs * Math.pow(2, attempt - 1);
    const cappedDelay = Math.min(exponentialDelay, this.maxDelayMs);
    
    // Add jitter: ±30%
    const jitter = cappedDelay * this.jitterFactor * (Math.random() * 2 - 1);
    return Math.round(cappedDelay + jitter);
  }

  shouldRetry(attempt, error) {
    if (attempt >= this.maxAttempts) return false;
    
    // Не ретраить определённые ошибки
    const nonRetryableErrors = ['AUTH_FAILED', 'RATE_LIMITED', 'BANNED'];
    if (error?.code && nonRetryableErrors.includes(error.code)) {
      return false;
    }
    
    return true;
  }
}

// Circuit Breaker
class CircuitBreaker {
  constructor(options = {}) {
    this.failureThreshold = options.failureThreshold || 5;
    this.recoveryTimeMs = options.recoveryTimeMs || 60000;
    this.failures = 0;
    this.lastFailureTime = 0;
    this.state = 'CLOSED'; // CLOSED, OPEN, HALF_OPEN
  }

  recordFailure() {
    this.failures++;
    this.lastFailureTime = Date.now();
    if (this.failures >= this.failureThreshold) {
      this.state = 'OPEN';
      console.warn(`[CIRCUIT] Opened after ${this.failures} failures`);
    }
  }

  recordSuccess() {
    this.failures = 0;
    this.state = 'CLOSED';
  }

  canExecute() {
    if (this.state === 'CLOSED') return true;
    
    if (this.state === 'OPEN') {
      if (Date.now() - this.lastFailureTime > this.recoveryTimeMs) {
        this.state = 'HALF_OPEN';
        console.log('[CIRCUIT] Half-open, allowing test request');
        return true;
      }
      return false;
    }
    
    return true; // HALF_OPEN
  }
}
Применение в background.js:
const llmCircuitBreakers = new Map(); // llmName → CircuitBreaker
const retryStrategy = new RetryStrategy({ maxAttempts: 4 });

async function dispatchWithRetry(llmName, tabId, prompt, attachments) {
  // Проверить circuit breaker
  if (!llmCircuitBreakers.has(llmName)) {
    llmCircuitBreakers.set(llmName, new CircuitBreaker());
  }
  const circuit = llmCircuitBreakers.get(llmName);
  
  if (!circuit.canExecute()) {
    console.warn(`[DISPATCH] ${llmName} circuit open, skipping`);
    return { success: false, reason: 'CIRCUIT_OPEN' };
  }

  let attempt = 0;
  while (true) {
    attempt++;
    try {
      const result = await dispatchPromptToTab(llmName, tabId, prompt, attachments);
      circuit.recordSuccess();
      return result;
    } catch (error) {
      circuit.recordFailure();
      
      if (!retryStrategy.shouldRetry(attempt, error)) {
        console.error(`[DISPATCH] ${llmName} giving up after ${attempt} attempts`);
        throw error;
      }
      
      const delay = retryStrategy.getDelay(attempt);
      console.log(`[DISPATCH] ${llmName} attempt ${attempt} failed, retry in ${delay}ms`);
      await sleep(delay);
    }
  }
}

// Оптимизированная активация вкладки
async function activateTabMinimal(tabId, llmName) {

    // Пользователь в другом приложении — переключаем тихо
    await chrome.tabs.update(tabId, { active: true });

    // Браузер в фокусе — минимизируем disruption

    const pendingDispatches = countPendingDispatches();

        await chrome.tabs.update(currentTab.id, { active: true });
      }
    }
  }
}

const STATES = {
  IDLE: 'IDLE',
  QUEUED: 'QUEUED',
  ACTIVATING: 'ACTIVATING',
  TYPING: 'TYPING',
  SUBMITTING: 'SUBMITTING',
  WAITING: 'WAITING',
  STREAMING: 'STREAMING',
  DONE: 'DONE',
  ERROR: 'ERROR'
};

const TRANSITIONS = {
  [STATES.IDLE]: [STATES.QUEUED],
  [STATES.QUEUED]: [STATES.ACTIVATING, STATES.ERROR],
  [STATES.ACTIVATING]: [STATES.TYPING, STATES.ERROR],
  [STATES.TYPING]: [STATES.SUBMITTING, STATES.ERROR],
  [STATES.SUBMITTING]: [STATES.WAITING, STATES.ERROR],
  [STATES.WAITING]: [STATES.STREAMING, STATES.DONE, STATES.ERROR],
  [STATES.STREAMING]: [STATES.DONE, STATES.ERROR],
  [STATES.DONE]: [STATES.IDLE],
  [STATES.ERROR]: [STATES.IDLE, STATES.QUEUED] // retry
};

class DispatchStateMachine {
  constructor(llmName) {
    this.llmName = llmName;
    this.state = STATES.IDLE;
    this.history = [];
    this.data = {};
    this.listeners = [];
  }

  canTransition(newState) {
    return TRANSITIONS[this.state]?.includes(newState);
  }

  transition(newState, data = {}) {
    if (!this.canTransition(newState)) {
      console.error(`[STATE] Invalid transition: ${this.state} → ${newState}`);
      return false;
    }

    const prev = this.state;
    this.state = newState;
    this.data = { ...this.data, ...data };
    this.history.push({
      from: prev,
      to: newState,
      timestamp: Date.now(),
      data
    });

    console.log(`[STATE] ${this.llmName}: ${prev} → ${newState}`);
    this.listeners.forEach(fn => fn(this.state, prev, data));
    return true;
  }

  onTransition(callback) {
    this.listeners.push(callback);
  }

  is(state) {
    return this.state === state;
  }

  isTerminal() {
    return this.state === STATES.DONE || this.state === STATES.ERROR;
  }

  reset() {
    this.state = STATES.IDLE;
    this.data = {};
  }
}

// Manager для всех моделей
const dispatchMachines = new Map();

function getDispatchMachine(llmName) {
  if (!dispatchMachines.has(llmName)) {
    dispatchMachines.set(llmName, new DispatchStateMachine(llmName));
  }
  return dispatchMachines.get(llmName);
}

23. Hybrid Mode (API + Web)
Отдельный проект. Требования:
    • UI для ввода/хранения API ключей разных провайдеров 
    • Fallback логика: API → Web если ключа нет 
    • Обработка ошибок API (rate limits, billing)
