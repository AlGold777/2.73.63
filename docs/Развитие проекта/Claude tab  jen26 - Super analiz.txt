КРИТИЧЕСКАЯ КОРРЕКЦИЯ МОИХ РЕШЕНИЙ
Почему мои A/B/C решения НЕДОСТАТОЧНЫ:
Мои решения фиксили только sleep() → visibility-aware. Но если:
    1. Session изменился пока sleep ждал → callback всё равно выстрелит в новый job 
    2. TabMapManager показывает старый tabId → dispatch уйдёт не туда 
    3. SPA Navigation произошла → content script работает с detached DOM 
Troттлинг — это симптом, не причина. Даже если sleep работает корректно, архитектурные race conditions всё равно ломают систему.

2.2. Visibility-Aware Sleep в content-utils.js
javascript
// content-utils.js — ПОЛНАЯ ЗАМЕНА sleep()
const sleep = (ms) => {
  return new Promise((resolve) => {
    if (ms <= 0) { resolve(); return; }
    
    const startedAt = Date.now();
    const speedMode = !!window.__PRAGMATIST_SPEED_MODE;
    const factor = speedMode ? 0.35 : 1;
    const targetMs = Math.max(speedMode ? 25 : 0, ms * factor);
    
    let timerId = null;
    let pausedAt = null;
    let resolved = false;
    
    const cleanup = () => {
      if (timerId) clearTimeout(timerId);
      document.removeEventListener('visibilitychange', handleVisibility);
    };
    
    const finish = () => {
      if (resolved) return;
      resolved = true;
      cleanup();
      resolve();
    };
    
    const scheduleRemaining = () => {
      if (resolved) return;
      
      // Рассчитываем сколько осталось (не считая время паузы)
      const activeTime = pausedAt 
        ? (pausedAt - startedAt) 
        : (Date.now() - startedAt);
      const remaining = Math.max(0, targetMs - activeTime);
      
      if (remaining <= 0) {
        finish();
        return;
      }
      
      if (document.visibilityState !== 'visible') {
        // Невидимы — запоминаем момент и ждём
        if (!pausedAt) {
          pausedAt = Date.now();
          requestFocusFromBackground();
        }
        return;
      }
      
      // Видимы — сбрасываем паузу и ставим таймер
      pausedAt = null;
      timerId = setTimeout(finish, remaining);
    };
    
    const handleVisibility = () => {
      if (resolved) return;
      
      if (document.visibilityState === 'visible') {
        scheduleRemaining();
      } else {
        if (timerId) clearTimeout(timerId);
        timerId = null;
        if (!pausedAt) pausedAt = Date.now();
      }
    };
    
    document.addEventListener('visibilitychange', handleVisibility);
    scheduleRemaining();
  });
};

// Throttled запрос фокуса у background
let lastFocusRequest = 0;
function requestFocusFromBackground() {
  const now = Date.now();
  if (now - lastFocusRequest < 3000) return;
  lastFocusRequest = now;
  
  try {
    chrome.runtime.sendMessage({ 
      type: 'NEED_FOCUS', 
      model: window.__LLM_MODEL__,
      reason: 'sleep_paused'
    }, () => chrome.runtime.lastError);
  } catch (_) {}
}

// spa-navigation-guard.js — загружается ПЕРВЫМ в content scripts
(function initSPANavigationGuard() {
  'use strict';
  
  if (window.__spaGuardInitialized) return;
  window.__spaGuardInitialized = true;
  
  let lastPathname = location.pathname;
  let lastHref = location.href;
  
  // History API interception
  const originalPushState = history.pushState;
  const originalReplaceState = history.replaceState;
  
  const handleNavigation = (oldUrl, newUrl) => {
    console.log(`[SPA-GUARD] Navigation: ${oldUrl} -> ${newUrl}`);
    
    // Cleanup старых ресурсов
    if (window.__currentCleanupScope?.cleanup) {
      try { window.__currentCleanupScope.cleanup('spa-navigation'); } catch (_) {}
    }
    
    // Уведомить background
    try {
      chrome.runtime.sendMessage({
        type: 'SPA_NAVIGATION',
        llmName: window.__LLM_MODEL__,
        oldUrl,
        newUrl
      }, () => chrome.runtime.lastError);
    } catch (_) {}
    
    // Реинициализация через задержку
    setTimeout(() => {
      if (typeof window.__reinitContentScript === 'function') {
        window.__reinitContentScript();
      }
    }, 500);
  };
  
  const checkNavigation = () => {
    if (location.pathname !== lastPathname) {
      const oldUrl = lastHref;
      lastPathname = location.pathname;
      lastHref = location.href;
      handleNavigation(oldUrl, lastHref);
    }
  };
  
  history.pushState = function(...args) {
    originalPushState.apply(this, args);
    checkNavigation();
  };
  
  history.replaceState = function(...args) {
    originalReplaceState.apply(this, args);
    checkNavigation();
  };
  
  window.addEventListener('popstate', checkNavigation);
  
  // Fallback: periodic check
  setInterval(checkNavigation, 1000);
})();
3.2. Extension Context Guard (добавить в content-utils.js)
javascript
// Extension Context Guard
const ExtensionContext = {
  _isValid: true,
  
  isValid() {
    if (!this._isValid) return false;
    try {
      return !!chrome.runtime?.id;
    } catch {
      this._isValid = false;
      this._handleInvalidation();
      return false;
    }
  },
  
  _handleInvalidation() {
    console.warn('[EXTENSION] Context invalidated — cleaning up');
    
    // Cleanup
    if (window.__currentCleanupScope?.cleanup) {
      try { window.__currentCleanupScope.cleanup('extension-invalidated'); } catch (_) {}
    }
    
    // Remove markers
    delete window.__llmCodexVersion;
    delete window.__spaGuardInitialized;
    
    console.info('[LLM_Codex] Extension reloaded. Please refresh this page.');
  }
};

// Safe message wrapper
function safeSendMessage(message, callback) {
  if (!ExtensionContext.isValid()) return false;
  
  try {
    chrome.runtime.sendMessage(message, (response) => {
      if (chrome.runtime.lastError?.message?.includes('Extension context invalidated')) {
        ExtensionContext._handleInvalidation();
        return;
      }
      if (callback) callback(response);
    });
    return true;
  } catch (e) {
    if (e.message?.includes('Extension context invalidated')) {
      ExtensionContext._handleInvalidation();
    }
    return false;
  }
}

window.ExtensionContext = ExtensionContext;
window.safeSendMessage = safeSendMessage;
