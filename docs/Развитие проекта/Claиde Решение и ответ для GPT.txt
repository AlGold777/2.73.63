  /**
   * Visibility-aware sleep that pauses when tab is hidden
   * and requests focus from background if needed.
   * Protects against Chrome Background Tab Throttling.
   */
  const sleep = (ms) => {
    const baseMs = Number(ms) || 0;
    const speedMode = !!window.__PRAGMATIST_SPEED_MODE;
    const factor = speedMode ? 0.35 : 1;
    const minMs = speedMode ? 25 : 0;
    const targetMs = Math.max(minMs, baseMs * factor);

    return new Promise((resolve) => {
      if (targetMs <= 0) {
        resolve();
        return;
      }

      const startedAt = Date.now();
      let timerId = null;
      let pausedAt = null;
      let resolved = false;

      const cleanup = () => {
        if (timerId) clearTimeout(timerId);
        document.removeEventListener('visibilitychange', handleVisibility);
      };

      const finish = () => {
        if (resolved) return;
        resolved = true;
        cleanup();
        resolve();
      };

      const getActiveTime = () => {
        if (pausedAt) {
          // Время до паузы
          return pausedAt - startedAt;
        }
        return Date.now() - startedAt;
      };

      const scheduleRemaining = () => {
        if (resolved) return;

        // Если невидимы — ждём без таймера
        if (document.visibilityState !== 'visible') {
          if (!pausedAt) {
            pausedAt = Date.now();
            requestFocusFromBackground('sleep_paused');
          }
          return;
        }

        // Видимы — сбрасываем паузу
        pausedAt = null;

        const activeTime = getActiveTime();
        const remaining = Math.max(0, targetMs - activeTime);

        if (remaining <= 0) {
          finish();
          return;
        }

        timerId = setTimeout(finish, remaining);
      };

      const handleVisibility = () => {
        if (resolved) return;

        if (document.visibilityState === 'visible') {
          // Вернулись — продолжаем
          scheduleRemaining();
        } else {
          // Ушли в фон — останавливаем таймер, запоминаем момент
          if (timerId) {
            clearTimeout(timerId);
            timerId = null;
          }
          if (!pausedAt) {
            pausedAt = Date.now();
          }
        }
      };

      document.addEventListener('visibilitychange', handleVisibility);
      scheduleRemaining();
    });
  };

  /**
   * Request focus from background (throttled to avoid spam)
   */
  let lastFocusRequestAt = 0;
  const FOCUS_REQUEST_THROTTLE_MS = 3000;

  const requestFocusFromBackground = (reason = 'unknown') => {
    const now = Date.now();
    if (now - lastFocusRequestAt < FOCUS_REQUEST_THROTTLE_MS) return;
    lastFocusRequestAt = now;

    // Проверяем extension context
    try {
      if (!chrome.runtime?.id) return;
    } catch (_) {
      return;

      }, () => {
        // Игнорируем ошибки (extension может быть перезагружен)
        if (chrome.runtime.lastError) { /* ignore */ }
      });
    } catch (_) {
      // Extension context invalidated
    }
